def outlier(data, column):
    mean = data[column].mean()
    std = data[column].std()
    lowest = mean - (std * 1.5)
    highest = mean + (std * 1.5)
    print('ìµœì†Œ ê²½ê³„ê°’: ', lowest, 'ìµœëŒ€ê²½ê³„ê°’:', highest)
    outlier_index = data[column][(data[column]<lowest) | (data[column] > highest) ].index
    return outlier_index
