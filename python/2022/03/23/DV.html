import matplotlib.pyplot as plt  ## 파이썬 내에서 그래프 출력시 디테일한 옵션
import matplotlib as mpl ## 한글폰트

mpl.rc('font', family='Malgun Gothic')

sns.countplot(data=df1, x='type_of_contract')
